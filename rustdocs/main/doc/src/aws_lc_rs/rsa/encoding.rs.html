4/// [RFC 8017](https://www.rfc-editor.org/rfc/rfc8017.html)
5///
6/// PKCS #1: RSA Cryptography Specifications Version 2.2
7pub(in crate::rsa) mod rfc8017 {
8    use crate::aws_lc::{
9        EVP_PKEY_assign_RSA, EVP_PKEY_new, RSA_parse_private_key, RSA_public_key_from_bytes,
10        RSA_public_key_to_bytes, EVP_PKEY,
11    };
12    use crate::cbs;
13    use crate::error::{KeyRejected, Unspecified};
14    use crate::ptr::{DetachableLcPtr, LcPtr};
15    use std::ptr::null_mut;
16
17    /// DER encode a RSA public key to `RSAPublicKey` structure.
18    pub(in crate::rsa) fn encode_public_key_der(
19        pubkey: &LcPtr<EVP_PKEY>,
20    ) -> Result<Box<[u8]>, Unspecified> {
21        let mut pubkey_bytes = null_mut::<u8>();
22        let mut outlen: usize = 0;
23        if 1 != unsafe {
24            RSA_public_key_to_bytes(&mut pubkey_bytes, &mut outlen, *pubkey.get_rsa()?)
25        } {
26            return Err(Unspecified);
27        }
28        let pubkey_bytes = LcPtr::new(pubkey_bytes)?;
29        let pubkey_slice = unsafe { pubkey_bytes.as_slice(outlen) };
30        let pubkey_vec = Vec::from(pubkey_slice);
31        Ok(pubkey_vec.into_boxed_slice())
32    }
33
34    /// Decode a DER encoded `RSAPublicKey` structure.
35    #[inline]
36    pub(in crate::rsa) fn decode_public_key_der(
37        public_key: &[u8],
38    ) -> Result<LcPtr<EVP_PKEY>, KeyRejected> {
39        let rsa = DetachableLcPtr::new(unsafe {
40            RSA_public_key_from_bytes(public_key.as_ptr(), public_key.len())
41        })?;
42
43        let mut pkey = LcPtr::new(unsafe { EVP_PKEY_new() })?;
44
45        if 1 != unsafe { EVP_PKEY_assign_RSA(*pkey.as_mut(), *rsa) } {
46            return Err(KeyRejected::unspecified());
47        }
48
49        rsa.detach();
50
51        Ok(pkey)
52    }
53
54    /// Decodes a DER encoded `RSAPrivateKey` structure.
55    #[inline]
56    pub(in crate::rsa) fn decode_private_key_der(
57        private_key: &[u8],
58    ) -> Result<LcPtr<EVP_PKEY>, KeyRejected> {
59        let mut cbs = cbs::build_CBS(private_key);
60
61        let rsa = DetachableLcPtr::new(unsafe { RSA_parse_private_key(&mut cbs) })?;
62
63        let mut pkey = LcPtr::new(unsafe { EVP_PKEY_new() })?;
64
65        if 1 != unsafe { EVP_PKEY_assign_RSA(*pkey.as_mut(), *rsa) } {
66            return Err(KeyRejected::unspecified());
67        }
68
69        rsa.detach();
70
71        Ok(pkey)
72    }
73}

75/// [RFC 5280](https://www.rfc-editor.org/rfc/rfc5280.html)
76///
77/// Encodings that use the `SubjectPublicKeyInfo` structure.
78pub(in crate::rsa) mod rfc5280 {
79    use crate::aws_lc::{EVP_PKEY, EVP_PKEY_RSA, EVP_PKEY_RSA_PSS};
80    use crate::buffer::Buffer;
81    use crate::encoding::PublicKeyX509Der;
82    use crate::error::{KeyRejected, Unspecified};
83    use crate::ptr::LcPtr;
84
85    pub(in crate::rsa) fn encode_public_key_der(
86        key: &LcPtr<EVP_PKEY>,
87    ) -> Result<PublicKeyX509Der<'static>, Unspecified> {
88        let der = key.marshal_rfc5280_public_key()?;
89        Ok(PublicKeyX509Der::from(Buffer::new(der)))
90    }
91
92    pub(in crate::rsa) fn decode_public_key_der(
93        value: &[u8],
94    ) -> Result<LcPtr<EVP_PKEY>, KeyRejected> {
95        LcPtr::<EVP_PKEY>::parse_rfc5280_public_key(value, EVP_PKEY_RSA).or(
96            // Does anyone encode with this OID?
97            LcPtr::<EVP_PKEY>::parse_rfc5280_public_key(value, EVP_PKEY_RSA_PSS),
98        )
99    }
