15macro_rules! platform_binding {
16    ($platform:ident, $platform_crypto:ident) => {
17        #[cfg(all($platform, not(feature = "ssl"), not(use_bindgen_pregenerated)))]
18        use_bindings!($platform_crypto);
19    };
